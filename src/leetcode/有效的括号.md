给定一个只包括 '('，')'，'{'，'}'，'['，']' 的字符串，判断字符串是否有效。

有效字符串需满足：

左括号必须用相同类型的右括号闭合。
左括号必须以正确的顺序闭合。
注意空字符串可被认为是有效字符串。
```
/**
 * @param {string} s
 * @return {boolean}
 */
var isValid = function(s) {
    var sta = [], target = {};
    target['('] = ')';
    target['['] = ']';
    target['{'] = '}';
    for(var i = 0, len = s.length; i < len; i++) {
    if (!sta.length) {
        sta.push(s[i])
    }
    else {
        if (s[i] === target[sta[sta.length - 1]])
        sta.pop();
        else 
        sta.push(s[i]);
    }
    }
    // console.log(sta)

    return !sta.length;
};
	
console.log(isValid("{[]}"))
```
